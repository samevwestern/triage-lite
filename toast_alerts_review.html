Below is a snippet showcasing exactly how these toast-driven error states are handled in production code:

// Production Toast Handler block in App.tsx
const showToast = (msg: string) => {
  // Dispatches visually to custom react toast component overlays
  setToastMessage(msg);
  setShowToastActive(true);
  setTimeout(() => {
    setShowToastActive(false);
  }, 3000);
};

// Camera exception validation workflow
try {
  const image = await Camera.getPhoto({ quality: 80 });
} catch (err: any) {
  const errMsg = err?.message || "";
  if (errMsg.toLowerCase().includes("permission") || errMsg.toLowerCase().includes("denied")) {
    showToast("⚠️ Camera permission denied! Please enable Camera in iPhone Settings.");
  } else {
    showToast("⚠️ Camera access cancelled. Using manual photo picker fallback.");
  }
}
